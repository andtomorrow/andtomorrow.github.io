let func = (arg1, arg2, ...argN) => expression
  • 인자 arg1..argN을 받는 함수 func이 만들어짐
  • 함수 func은 화살표(=>) 우측의 표현식(expression)을 평가하고, 평가 결과를 반환함
  • 아래의 동일한 함수를 축약한 것
let func = function(arg1, arg2, ...argN) {
  return expression;
};

하나의 인수만을 갖는 경우, 괄호 생략하여 코드 길이를 더 줄일 수 있음

let double = n => n * 2;
// let double = function(n) { return n * 2}과 거의 동일함

alert( double(3) ); // 6

  • 화살표 함수는 일반 함수 표현식과 동일한 방식으로 사용할 수 있음

let age = prompt("나이를 알려주세요.", 18);

let welcome = (age < 18) ?
  () => alert('안녕') :
  () => alert("안녕하세요!");

welcome();
